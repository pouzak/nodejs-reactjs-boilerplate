$ mkdir ${HOME}/postgres-data/

sudo docker run -d --name dev-postgres -e POSTGRES_PASSWORD=12345678 -v /home/paulius/Desktop/docker/postgres-data/:/var/lib/postgresql/data -p 5432:5432 postgres
